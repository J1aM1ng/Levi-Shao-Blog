```typescript
// 贪心
// 排除不可能作为起点的点（失败路径上的点）
function canCompleteCircuit(gas: number[], cost: number[]): number {
    const len = gas.length;
    // 起点
    let start = 0;
    // 目前总剩油量
    let sum = 0;
    // 当前剩油量
    let currentSum = 0;
    for (let i = 0; i < len; i++) {
        sum += gas[i] - cost[i];
        currentSum += gas[i] - cost[i];
        if (currentSum < 0) {
            currentSum = 0;
            // 从失败路径之后开始寻起点
            start = i + 1;
        }    
    }
    if (sum < 0) {
        return -1;
    }
    return start;
};
```
