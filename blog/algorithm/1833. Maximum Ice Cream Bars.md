```typescript
// 排序+贪心
// 时间O(nlogn)
function maxIceCream(costs: number[], coins: number): number {
    // 升序
    costs.sort((x, y) => x - y);
    let ans = 0;
    let money = coins;
    for (const cost of costs) {
        if (cost <= money) {
            ++ans;
            money -= cost;
        } else {
            break;
        }
    }
    return ans;
};
```
