```typescript
// 滑动窗口 哈希表
const lengthOfLongestSubstring = (s: string): number => {
    const st = new Set();
    let ans = 0;
    let left = 0, right = 0, len = s.length;  
    while (left < len) {     
        while (right < len && !st.has(s[right])) {
            st.add(s[right]);
            // 右区间扩展
            ++right;
        }
        // 此时记录长度，然后收缩左指针
        ans = Math.max(st.size, ans);
        st.delete(s[left]);
        // 左区间内缩
        ++left;
    }
    return ans;
};
```
